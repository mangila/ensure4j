12  public class ObjectTest {

14      @Test
15      @DisplayName("Happy path equals(Object)")
16      void equals() {
17          assertThatCode(() -> Ensure.isEquals(new BigDecimal("1"), new BigDecimal("1"))).doesNotThrowAnyException();
18          assertThatCode(() -> Ensure.isEquals(1, 1, "test message")).doesNotThrowAnyException();
19          assertThatCode(() -> Ensure.isEquals(1, 1, () -> new IllegalArgumentException("test message"))).doesNotThrowAnyException();
20          // REMINDME: idk... ok?
21          assertThatCode(() -> Ensure.isEquals(null, null)).doesNotThrowAnyException();
22      }

24      @Test
25      @DisplayName("Sad path equals(Object)")
26      void equals1() {
27          assertThatThrownBy(() -> Ensure.isEquals(new BigDecimal("1"), new BigInteger("1")))
28                  .isInstanceOf(EnsureException.class)
29                  .hasMessage("objects must be equal");
30          assertThatThrownBy(() -> Ensure.isEquals(new BigDecimal("1"), new BigInteger("1"), "test objects not equal"))
31                  .isInstanceOf(EnsureException.class)
32                  .hasMessage("test objects not equal");
33          assertThatThrownBy(() -> Ensure.isEquals(new BigDecimal("1"), new BigInteger("1"), () -> new IllegalArgumentException("test objects not equal")))
34                  .isInstanceOf(IllegalArgumentException.class)
35                  .hasMessage("test objects not equal");
36          assertThatThrownBy(() -> Ensure.isEquals(null, new BigInteger("1"), () -> new IllegalArgumentException("test objects not equal")))
37                  .isInstanceOf(IllegalArgumentException.class)
38                  .hasMessage("test objects not equal");
39      }

41      @Test
42      @DisplayName("Happy path isInstanceOf(Class)")
43      void isInstanceOf() {
44          assertThatCode(() -> Ensure.isInstanceOf(String.class, "test"))
45                  .doesNotThrowAnyException();
46          assertThatCode(() -> Ensure.isInstanceOf(Integer.class, 1, "test message"))
47                  .doesNotThrowAnyException();
48          assertThatCode(() -> Ensure.isInstanceOf(Float.class, 32.00f, () -> new IllegalArgumentException("test message")))
49                  .doesNotThrowAnyException();
50      }

52      @Test
53      @DisplayName("Sad path isInstanceOf(Class)")
54      void isInstanceOf1() {
55          assertThatThrownBy(() -> Ensure.isInstanceOf(null, "test"))
56                  .isInstanceOf(EnsureException.class)
57                  .hasMessage("clazz must not be null");
58          assertThatThrownBy(() -> Ensure.isInstanceOf(null, "test", "test message"))
59                  .isInstanceOf(EnsureException.class)
60                  .hasMessage("test message");
61          assertThatThrownBy(() -> Ensure.isInstanceOf(null, "test", () -> new IllegalArgumentException("test message")))
62                  .isInstanceOf(IllegalArgumentException.class)
63                  .hasMessage("test message");
64          assertThatThrownBy(() -> Ensure.isInstanceOf(String.class, 'c'))
65                  .isInstanceOf(EnsureException.class)
66                  .hasMessage("object must be an instance of java.lang.String");
67          assertThatThrownBy(() -> Ensure.isInstanceOf(Integer.class, 2L, "test message"))
68                  .isInstanceOf(EnsureException.class)
69                  .hasMessage("test message");
70          assertThatThrownBy(() -> Ensure.isInstanceOf(Double.class, 2f, () -> new IllegalArgumentException("test message")))
71                  .isInstanceOf(IllegalArgumentException.class)
72                  .hasMessage("test message");
73      }
