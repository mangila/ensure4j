23      /**
24       * Checks if the given object is an instance of the specified class.
25       * If the object is not an instance of the class, a RuntimeException
26       * provided by the supplier is thrown.
27       *
28       * @param clazz    the class to check the object against
29       * @param object   the object to verify
30       * @param supplier a supplier that provides the exception to be thrown
31       *                 if the object is not an instance of the specified class
32       * @throws RuntimeException if the object is not an instance of the specified class
33       */
34      public static void isInstanceOf(Class<?> clazz, Object object, Supplier<RuntimeException> supplier) throws RuntimeException {
35          notNull(clazz, supplier);
36          if (!clazz.isInstance(object)) {
37              throw getSupplierOrThrow(supplier);
38          }
39      }

66      /**
67       * Compares two objects for equality and throws a provided exception if they are not equal.
68       * If both objects are the same instance or the first object equals the second, the method returns without exception.
69       * If the first object is null or the objects are not equal, a custom exception is thrown.
70       *
71       * @param object                   the first object to compare, must not be null
72       * @param otherObject              the second object to compare with the first one
73       * @param runtimeExceptionSupplier the runtimeExceptionSupplier providing the exception to be thrown if objects are not equal
74       * @throws RuntimeException if the first object is null or the objects are not equal
75       */
76      public static void isEquals(Object object, Object otherObject, Supplier<RuntimeException> runtimeExceptionSupplier) throws RuntimeException {
77          if (object == otherObject) {
78              return;
79          }
80          notNull(object, runtimeExceptionSupplier);
81          if (object.equals(otherObject)) {
82              return;
83          }
84          throw getSupplierOrThrow(runtimeExceptionSupplier);
85      }

573     /**
574      * Retrieves the value provided by the given {@link Supplier}. If the supplier is null or the supplied
575      * value is null, an {@link EnsureException} is thrown.
576      *
577      * @param <T>      the type of the object supplied by the {@code Supplier}
578      * @param supplier the {@link Supplier} to provide the value; must not be null and must not return null
579      * @return the non-null value provided by the {@code supplier}
580      * @throws EnsureException if the {@code supplier} is null or produces a null value
581      */
582     @NonNull
583     private static <T> T getSupplierOrThrow(Supplier<T> supplier) {
584         if (supplier == null) {
585             throw new EnsureException("supplier was null");
586         }
587         final T t = supplier.get();
588         if (t == null) {
589             throw new EnsureException("supplier was given a null value");
590         }
591         return t;
592     }
