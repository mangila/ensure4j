3   /**
4    * Represents an exception that can be used for validation or ensuring specific conditions are met
5    * during runtime. This exception is unchecked and extends {@code RuntimeException}.
6    * <br>
7    * The {@code EnsureException} provides a constructor to define a custom exception message and a
8    * static factory method for creating new instances using a provided message.
9    */
10  public class EnsureException extends RuntimeException {
11  
12      /**
13       * Constructs a new {@code EnsureException} with the specified detail message.
14       *
15       * @param message the detail message that provides information about the exception.
16       */
17      public EnsureException(String message) {
18          super(message);
19      }
20  
21      /**
22       * Creates a new instance of {@code EnsureException} with the specified message.
23       *
24       * @param message the detail message that describes the reason for the exception
25       * @return a new {@code EnsureException} instance encapsulating the provided message
26       */
27      public static EnsureException of(String message) {
28          return new EnsureException(message);
29      }
30  }
